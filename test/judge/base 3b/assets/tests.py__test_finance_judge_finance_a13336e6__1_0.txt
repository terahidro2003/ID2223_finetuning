/* Step 1: Calculate the overlapping log returns for each stock */
data stkcorr;
set TRD_Dalyr;
lagstk=lag(Dretwd);
lag2stk=lag(Dretwd);
run;

/* Step 2: Remove the first two data points for each stock */
data stkcorr2;
set stkcorr;
by stkcd;
if first.stkcd then delete;
run;

/* Step 3: Remove the first two data points for each stock */
data stkcorr3;
set stkcorr2;
by stkcd;
if first.stkcd then delete;
run;

/* Step 4: Calculate the overlapping returns for each stock */
data stkcorr4;
set stkcorr3;
dstk=log(1+Dretwd)+log(1+lagstk)+log(1+lag2stk);
drop Dretwd lagstk lag2stk;
run;

/* Step 5: Sort the data for stock correlation */
proc sort data=stkcorr4;
by Trddt stkcd;
run;

/* Step 6: Calculate the overlapping log returns for the market */
data mktcorr;
set fivefactor_daily;
mkt=mkt_rf+rf;
lagmkt=lag(mkt);
lag2mkt=lag2(mkt);
if missing(lagmkt)=0 and missing(lag2mkt)=0 then do;
  dmkt=log(1+mkt)+log(1+lagmkt)+log(1+lag2mkt);
end;
Trddt=trddy;
keep Trddt dmkt;
run;

/* Step 7: Merge the stock and market data and calculate correlation coefficients */
data corr;
merge stkcorr4 mktcorr;
by Trddt;
run;

/* Step 8: Prepare the data for correlation calculation */
data corr2;
set corr;
year=input(substr(left(Trddt),1,4),4.);
month=input(substr(left(Trddt),6,2),2.);
m=(year-1994)*12+month;
if missing(dstk) then delete;
if missing(dmkt) then delete;
drop Trddt year month;
run;

/* Step 9: Calculate correlation coefficients for each stock and the market */
data corrtotal;
input stkcd $ tmonth corr;
datalines;
'000000' 0 0
;
run;

%macro corr;
%do i=1 %to 279;
data corr3;
set corr2;
if m>&i-60 and m<=&i;
tmonth=&i;
run;
proc sort data=corr3;
by stkcd tmonth;
run;
proc corr data=corr3 out=corr4 noprint;
var dstk dmkt;
by stkcd tmonth;
run;
data corr5;
set corr4;
if _TYPE_='MEAN' then delete;
if _TYPE_='STD' then delete;
if _TYPE_='CORR' and _NAME_='dstk' then do;
corr=dmkt;
end;
if _TYPE_='CORR' and _NAME_='dmkt' then delete;
drop _NAME_;
run;
data corr6;
set corr5;
lagnstk=lag(dstk);
lagnmkt=lag(dmkt);
run;
data corr7;
set corr6;
if _TYPE_='N' then delete;
if _TYPE_='CORR' and lagnstk<750 then delete;
if _TYPE_='CORR' and lagnmkt<750 then delete;
drop _TYPE_ dstk dmkt lagnstk lagnmkt;
run;
data corrtotal;
set corrtotal corr7;
run; 
%end;
%mend corr;
%corr;
data corrtotal;
set corrtotal;
if tmonth=0 then delete;
run;
data corrresult4;
set corrtotal;
run;
proc sort data=corrresult4;

run;
