def min_operations(a, b):
    operations = 0
    while a != b:
        if a % 2 == 0 and b % 2 == 0:
            if b % 4 == 0:
                a = a * 4
                operations += 1
            elif b % 8 == 0:
                a = a * 8
                operations += 1
            else:
                a = a * 2
                operations += 1
        elif a % 4 == 0 and b % 2 == 0:
            if b % 8 == 0:
                a = a * 4
                operations += 1
            else:
                a = a * 2
                operations += 1
        elif a % 8 == 0 and b % 4 == 0:
            if b % 8 == 0:
                a = a * 4
                operations += 1
            else:
                a = a * 2
                operations += 1
        else:
            a = a // 2
            operations += 1
    return operations

t = int(input())
for _ in range(t):
    a, b = map(int, input().split())
    if min_operations(a, b) == 0:
        print(0)
    elif min_operations(a, b) > 100:
        print(-1)
    else:
        print(min_operations(a, b))
