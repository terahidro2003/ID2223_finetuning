import random
import math

def estimate_pi(num_points):

        x = random.uniform(0, 1)
        y = random.uniform(0, 1)

        # Check if the point is within the quarter circle
        distance = math.sqrt(x**2 + y**2)
        if distance <= 1:
            points_inside_circle += 1
        else:
            points_outside_circle += 1

    # Calculate the ratio of points within the quarter circle to the total number of points
    ratio = points_inside_circle / num_points

    # Estimate the value of π using the ratio

    return estimated_pi
