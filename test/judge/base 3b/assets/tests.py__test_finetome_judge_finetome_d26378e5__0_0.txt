def pathCounts(root):

            return 0
        paths = helper(node.left, path_len + 1) + helper(node.right, path_len + 1)
        if node.left is None and node.right is None:
            paths += 1
        return paths

    if root is None:
        return ""
    else:

        for i in range(1, paths + 1):
            path_len = i
            paths_i = helper(root, path_len)
            path_str += str(paths_i) + " $"
        return path_str
