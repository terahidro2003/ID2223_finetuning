const normalizeData = (columns, method) => {
  const normalizedData = JSON.parse(JSON.stringify(columns)); // Create a deep copy of the original data

  for (const column of normalizedData) {
    if (column === 'salary' || column === 'experience') {
      const min = normalizedData[column].reduce((a, b) => Math.min(a, b), Infinity);
      const max = normalizedData[column].reduce((a, b) => Math.max(a, b), -Infinity);
      const range = max - min;

      for (let i = 0; i < normalizedData[column].length; i++) {
        const value = normalizedData[column][i];
        normalizedData[column][i] = (value - min) / range;
      }
    }
  }

  return normalizedData;
};

const columns = [
  { name: 'salary', values: [50000, 60000, 70000, 80000, 90000] },
  { name: 'experience', values: [2, 5, 8, 10, 12] },
  { name: 'name', values: ['John', 'Jane', 'Bob', 'Alice', 'Mike'] },
];

const normalizedColumns = normalizeData(columns, 'min-max');

console.log(normalizedColumns);
